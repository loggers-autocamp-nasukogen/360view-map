    window.addEventListener("DOMContentLoaded", function () {
      const video = document.getElementById("main-video");
      if (video) {

      function generateWavePath(width, height, segments, amplitude, frequency, phase = 0) {
        const step = width / segments;
        let path = `M0 ${height / 2}`;
        for (let i = 0; i <= segments; i++) {
          const x = i * step;
          const y = height / 2 + Math.sin((i / segments) * frequency * Math.PI * 2 + phase) * amplitude;
          path += ` L${x} ${y}`;
        }
        path += ` L${width} ${height} L0 ${height} Z`;
        return path;
      }

      const width = 1920;
      const height = 100;
      const segments = 80;
      const amplitude = 10;
      const frequency = 1.2;

      const path1 = document.getElementById("wave-path-1");
      const path2 = document.getElementById("wave-path-2");

      let phase = 0;
      function animateWave() {
        phase += 0.015;
        const d = generateWavePath(width, height, segments, amplitude, frequency, phase);
        path1.setAttribute("d", d);
        path2.setAttribute("d", d);
        requestAnimationFrame(animateWave);
      }

      animateWave();
    });
